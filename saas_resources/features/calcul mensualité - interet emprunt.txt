L = loan amount
m = mensuality
r = nominal rate
i = interet
n = duration in year / month

m = (L * (r/12)) / (1 - (1 + (r/12)) ^ -n) // ici n est la durée en mois
i = 12 * n * m - L // ici n est la durée en année ex: for 6 months -> 0.5
total payment = m * n
loan cost = total payment - L
